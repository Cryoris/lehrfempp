398  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
399 
400  // Index counter for global shape functions = global dof
401  gdof_idx_t dof_idx = 0;
402 
403  // Step I: Set indices for shape functions on nodes
404  // Run through node indices (entities of co-dimension 2)
405  const size_type no_nodes = mesh_p_->Size(2);
406  no_int_dofs_[2].resize(no_nodes, 0);
407  offsets_[2].resize(no_nodes + 1, 0);
408  // Traverse nodes (co-dimension-2 entities) based on indices
409  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
410  // Obtain pointer to node entity
411  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
412  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
413  // Offset for indices of node in index vector
414  glb_idx_t node_dof_offset = dof_idx;
415  offsets_[2][node_idx] = node_dof_offset;
416  // Request number of local shape functions associated with the node
417  size_type no_int_dof_node = locdof(*node_p);
418  no_int_dofs_[2][node_idx] = no_int_dof_node;
419 
420  // Store dof indices in array
421  for (int j = 0; j < no_int_dof_node; j++) {
422  dofs_[2].push_back(dof_idx);
423  dof_entities_.push_back(node_p); // Store entity for current dof
424  dof_idx++; // Move on to next index
425  }
426  }
427  // Set sentinel
428  offsets_[2][no_nodes] = dof_idx;
429 
430  // Step II: Set indices for shape functions on edges (co-dimension = 1)
431  const size_type no_edges = mesh_p_->Size(1);
432  // Set length of edge-related index vectors
433  no_int_dofs_[1].resize(no_edges, 0);
434  offsets_[1].resize(no_edges + 1, 0);
435  // points to the position of the current dof index
436  size_type edge_dof_offset = 0;
437  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
438  // Obtain pointer to edge entity
439  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
440  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
441  // Offset for indices of edge dof in index vector
442  offsets_[1][edge_idx] = edge_dof_offset;
443  size_type no_int_dof_edge = locdof(*edge);
444  no_int_dofs_[1][edge_idx] = no_int_dof_edge;
445 
446  // Obtain indices for basis functions sitting at endpoints
447  // Endpoints are mesh entities with co-dimension = 2
448  for (const lf::mesh::Entity &endpoint : edge->SubEntities(1)) {
449  const glb_idx_t ep_idx(mesh_p_->Index(endpoint));
450  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
451  size_type no_int_dofs_ep = no_int_dofs_[2][ep_idx];
452  // Copy indices of shape functions from nodes to edge
453  for (int j = 0; j < no_int_dofs_ep; j++) {
454  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
455  edge_dof_offset++;
456  }
457  }

459  for (int j = 0; j < no_int_dof_edge; j++) {
460  dofs_[1].push_back(dof_idx);
461  edge_dof_offset++;
462  dof_entities_.push_back(edge);
463  dof_idx++;
464  }
465  }

467  offsets_[1][no_edges] = edge_dof_offset;

474  // points to the position of the current dof index

476  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
477  // Obtain pointer to current ell
478  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
479  // Offset for indices of cell dof in index vector
480  offsets_[0][cell_idx] = cell_dof_offset;
481  size_type no_int_dof_cell = locdof(*cell);
482  no_int_dofs_[0][cell_idx] = no_int_dof_cell;
483 
484  // Obtain indices for basis functions in vertices
485  for (const lf::mesh::Entity &vertex : cell->SubEntities(2)) {
486  const glb_idx_t vt_idx(mesh_p_->Index(vertex));
487  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
488  size_type no_int_dofs_vt = no_int_dofs_[2][vt_idx];
489  // Copy indices of shape functions from nodes to cell
490  for (int j = 0; j < no_int_dofs_vt; j++) {
491  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
492  cell_dof_offset++;
493  }
494  }
495

497  // Internal ordering will depend on the orientation of the edge
498  lf::base::RandomAccessRange<const lf::mesh::Orientation>
499  edge_orientations(cell->RelativeOrientations());
500  lf::base::RandomAccessRange<const lf::mesh::Entity> edges(
501  cell->SubEntities(1));

503  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
504  ed_sub_idx++) {
505  const glb_idx_t edge_idx = mesh_p_->Index(edges[ed_sub_idx]);
506  const size_type no_int_dof_edge = no_int_dofs_[1][edge_idx];
507  const glb_idx_t edge_int_dof_offset =
508  offsets_[1][edge_idx + 1] - no_int_dof_edge;

511  // The order, in which they are copied depends on the relative
512  // orientation of the edge w.r.t. the cell
513  switch (edge_orientations[ed_sub_idx]) {
514  case lf::mesh::Orientation::positive: {

516  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
517  cell_dof_offset++;
518  }
519  break;
520  }
521  case lf::mesh::Orientation::negative: {
522  for (int j = no_int_dof_edge - 1; j >= 0; j--) {
523  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
524  cell_dof_offset++;
525  }
526  break;
527  }
528  } // end switch
529  } // end loop over edges

533  dofs_[0].push_back(dof_idx);
534  cell_dof_offset++;

537  } // end loop over interior dof of cell
538  } // end loop over cells
539  // Set sentinel

544  } // end constructor
