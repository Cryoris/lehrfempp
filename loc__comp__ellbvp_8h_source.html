439 typename LagrangeFEEllBVPElementMatrix<DIFF_COEFF, REACTION_COEFF>::ElemMat
440 LagrangeFEEllBVPElementMatrix<DIFF_COEFF, REACTION_COEFF>::Eval(
441  const lf::mesh::Entity &cell) {
442  // Type for diffusion coefficient
443  using diff_coeff_t = decltype(alpha_(Eigen::Vector2d::Zero()));
444  // Type for reaction coefficient
445  using reac_coeff_t = decltype(gamma_(Eigen::Vector2d::Zero()));
446 
447  // Topological type of the cell
448  const lf::base::RefEl ref_el{cell.RefEl()};
449  // Query the shape of the cell
450  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
451  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
452  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
453  "Only 2D implementation available!");
454  SWITCHEDSTATEMENT(ctrl_, kout_cell,
455  std::cout << ref_el << ", shape = \n"
456  << geo_ptr->Global(ref_el.NodeCoords())
457  << std::endl);
458 
459  // Computations differ depending on the type of the cell
460  switch (ref_el) {
461  case lf::base::RefEl::kTria(): {
462  // Quadrature points in actual cell
463  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_tria_.Points()));
464  // Obtain the metric factors for the quadrature points
465  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_tria_.Points()));
466  // Fetch the transformation matrices for the gradients
467  const Eigen::MatrixXd JinvT(geo_ptr->JacobianInverseGramian(qr_tria_.Points()));
468 
469  // Element matrix
470  elem_mat_t mat(Nrsf_tria_, Nrsf_tria_);
471  mat.setZero();
472 
473  // Loop over quadrature points
474  for (int k = 0; k < Nqp_tria_; ++k) {
475  // Evaluate diffusion and reaction coefficient
476  // at quadrature points in actual cell
477  const auto alphaval = alpha_(mapped_qpts.col(k));
478  const auto gammaval = gamma_(mapped_qpts.col(k));
479 
480  const double w = qr_tria_.Weights()[k] * determinants[k];
481  // Transformed gradients
482  const auto trf_grad(JinvT.block(0, 2 * k, 2, 2) *
483  grad_quadpoint_tria_[k]);
484  // Transformed gradients multiplied with coefficient
485  const auto alpha_trf_grad(alphaval * trf_grad);
486  mat += w * (alpha_trf_grad.transpose() * trf_grad +
487  (gammaval * rsf_quadpoints_tria_.col(k)) *
488  (rsf_quadpoints_tria_.col(k).transpose()));
489  }
490  return mat;
491  break;
492  }
493  case lf::base::RefEl::kQuad(): {
494  // Quadrature points in actual cell
495  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_quad_.Points()));
496  // Obtain the metric factors for the quadrature points
497  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_quad_.Points()));
498  // Fetch the transformation matrices for the gradients
499  const Eigen::MatrixXd JinvT(geo_ptr->JacobianInverseGramian(qr_quad_.Points()));
500 
501  // Element matrix
502  elem_mat_t mat(Nrsf_quad_, Nrsf_quad_);
503  mat.setZero();
504 
505  // Loop over quadrature points
506  for (int k = 0; k < Nqp_quad_; ++k) {
507  // Evaluate diffusion and reaction coefficient
508  // at quadrature points in actual cell
509  const auto alphaval{alpha_(mapped_qpts.col(k))};
510  const auto gammaval{gamma_(mapped_qpts.col(k))};
511 
512  const double w = qr_quad_.Weights()[k] * determinants[k];
513  // Transformed gradients
514  const auto trf_grad(JinvT.block(0, 2 * k, 2, 2) *
515  grad_quadpoint_quad_[k]);
516  // Transformed gradients multiplied with coefficient
517  const auto alpha_trf_grad(alphaval * trf_grad);
518  mat += w * (alpha_trf_grad.transpose() * trf_grad +
519  (gammaval * rsf_quadpoints_quad_.col(k)) *
520  (rsf_quadpoints_quad_.col(k).transpose()));
521  }
522  return mat;
523  break;
524  }
525  default: { LF_ASSERT_MSG(false, "Illegal cell type"); }
526  } // end switch

705 ScalarFELocalLoadVector<SCALAR, FUNCTOR>::Eval(const lf::mesh::Entity &cell) {
706  // Type for source function
707  using source_fn_t = decltype(f_(Eigen::Vector2d::Zero()));
708  // Topological type of the cell
709  const lf::base::RefEl ref_el{cell.RefEl()};
710  // Query the shape of the cell
711  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
712  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
713  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
714  "Only 2D implementation available!");
715  SWITCHEDSTATEMENT(ctrl_, kout_cell,
716  std::cout << ref_el << ", shape = \n"
717  << geo_ptr->Global(ref_el.NodeCoords())
718  << std::endl);
719 
720  // Computations differ depending on the type of the cell
721  switch (ref_el) {
722  case lf::base::RefEl::kTria(): {
723  // World coordinates of quadrature points
724  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_tria_.Points()));
725  SWITCHEDSTATEMENT(ctrl_, kout_qpts,
726  std::cout << "LOCVEC(Tria): Mapped quadrature points:\n"
727  << mapped_qpts << std::endl);
728  // Obtain the metric factors for the quadrature points
729  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_tria_.Points()));
730  SWITCHEDSTATEMENT(ctrl_, kout_dets,
731  std::cout << "LOCVEC(Tria): Metric factors:\n"
732  << determinants.transpose() << std::endl);
733  // Element vector
734  elem_vec_t vec(Nrsf_tria_);
735  vec.setZero();
736 
737  // Loop over quadrature points
738  for (int k = 0; k < Nqp_tria_; ++k) {
739  // Source function value at quadrature point
740  const auto fval = f_(mapped_qpts.col(k));
741  SWITCHEDSTATEMENT(
742  ctrl_, kout_loop,
743  std::cout << "LOCVEC(Tria): ["
744  << qr_tria_.Points().col(k).transpose() << "] -> ["
745  << mapped_qpts.col(k).transpose() << "], f = " << fval
746  << ", weight = " << qr_tria_.Weights()[k] << std::endl);
747  // Contribution of current quadrature point
748  vec +=
749  (qr_tria_.Weights()[k] * determinants[k] * fval) * rsf_quadpoints_tria_.col(k);
750  }
751  SWITCHEDSTATEMENT(ctrl_, kout_locvec,
752  std::cout << "LOCVEC(Tria) = \n"
753  << vec.transpose() << std::endl);
754  return vec;
755  break;
756  }
757  case lf::base::RefEl::kQuad(): {
758  // Quadrature points in world coordinates
759  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_quad_.Points()));
760  SWITCHEDSTATEMENT(ctrl_, kout_qpts,
761  std::cout << "LOCVEC(Quad): Mapped quadrature points:\n"
762  << mapped_qpts << std::endl);
763  // Obtain the metric factors for the quadrature points
764  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_quad_.Points()));
765  SWITCHEDSTATEMENT(ctrl_, kout_dets,
766  std::cout << "LOCVEC(Quad): Metric factors:\n"
767  << determinants.transpose() << std::endl);
768  // Element vector
769  elem_vec_t vec(Nrsf_quad_);
770  vec.setZero();
771 
772  // Loop over quadrature points
773  for (int k = 0; k < Nqp_quad_; ++k) {
774  // Source function value at quadrature point
775  const auto fval = f_(mapped_qpts.col(k));
776  SWITCHEDSTATEMENT(
777  ctrl_, kout_loop,
778  std::cout << "LOCVEC(Quad): ["
779  << qr_quad_.Points().col(k).transpose() << "] -> ["
780  << mapped_qpts.col(k).transpose() << "], f = " << fval
781  << ", weight = " << qr_quad_.Weights()[k] << std::endl);
782  // Contribution of current quadrature point
783  vec +=
784  (qr_quad_.Weights()[k] * determinants[k] * fval) * rsf_quadpoints_quad_.col(k);
785  }
786  SWITCHEDSTATEMENT(ctrl_, kout_locvec,
787  std::cout << "LOCVEC(Quad) = \n"
788  << vec.transpose() << std::endl);
789  return vec;
790  break;
791  }
792  default: { LF_ASSERT_MSG(false, "Illegal cell type"); }
793  } // end switch
